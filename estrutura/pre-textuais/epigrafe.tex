\renewcommand{\epigraphname}{EPÍGRAFE}

\begin{epigrafe}

\textit{Um barulho no quarto\\
Um susto, um rato\\
Da minha cama eu avisto\\
Livros, teias, traças\\
O canto negro de um pássaro}

\hspace{7em} \textit{(Rogério Skylab, "O Corvo")
}
\end{epigrafe}
